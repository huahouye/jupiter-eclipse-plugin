005    import java.util.List;

029    /**
030     * Provides ReviewIssue ID setting page.
031     * 
032     * @author Takuya Yamashita
033     * @version $Id: ReviewIdNewReviewIdPage.java,v 1.4 2005/03/25 18:32:06 takuyay Exp $
034     */
035    public class ReviewIdNewReviewIdPage extends WizardPage {
036      private IWorkbench workbench;
037      private Combo projectCombo;
038      private Text reviewIdText;
039      private Label infoLabel;
040      private List tempReviewIdList = new ArrayList();
041      private Text descriptionText;
042      private IProject project;
043    
044      /**
045       * Instantiates the config review id page.
046       * @param workbench the workbench
047       * @param pageName the page name.
048       * @param project the project.
049       * @param imageFilePath the image file path.
050       */
051      public ReviewIdNewReviewIdPage(IWorkbench workbench, String pageName, IProject project, 
052                                         String imageFilePath) {
053        super(pageName);
054        setImageDescriptor(ReviewPlugin.createImageDescriptor(imageFilePath));
055        setTitle(ReviewI18n.getString("ReviewIdNewReviewIdPage.label.title.before")
056                 + project.getName()
057                 + ReviewI18n.getString("ReviewIdNewReviewIdPage.label.title.after"));
058        String message = ReviewI18n.getString("ReviewIdNewReviewIdPage.label.title.description");
059        applyToStatusLine(new Status(IStatus.OK, "not_used", 0, message , null));
060        this.workbench = workbench;
061        this.project = project;
062      }
063    
064      /**
065       * Creates control.
066       * @see org.eclipse.jface.dialogs.IDialogPage#createControl(org.eclipse.swt.widgets.Composite)
067       */
068      public void createControl(Composite ancestor) {
069        Composite parent = createsGeneralComposite(ancestor);
070        createLabelContent(parent);
071        createReviewIdContent(parent);
072        createDescriptionContent(parent);
073        //  set the composite as the control for this page
074        setControl(parent);
075        addListeners();
076        setPageComplete(isTextNonEmpty(reviewIdText) && isTextNonEmpty(descriptionText));
077      }
078    
079      /**
080       * Creates view preference frame and return the child composite.
081       * 
082       * @param parent the parent composite.
083       * @return the child composite.
084       */
085      private Composite createsGeneralComposite(Composite parent) {
086        Composite child = new Composite(parent, SWT.LEFT);
087        FormLayout layout = new FormLayout();
088        layout.marginWidth = 7;
089        layout.marginHeight = 7;
090        child.setLayout(layout);
091        return child;
092      }
093      
094      /**
095       * Creates label content.
096       * @param parent the parent composite.
097       */
098      private void createLabelContent(Composite parent) {
099        this.infoLabel = new Label(parent, SWT.NULL);
100        infoLabel.setText(ReviewI18n.getString("ReviewIdNewReviewIdPage.label.info"));
101      }
102    
103      /**
104       * Creates review ID content.
105       * 
106       * @param parent the parent.
107       */
108      private void createReviewIdContent(Composite parent) {
109        Label reviewIdLabel = new Label(parent, SWT.NONE);
110        reviewIdLabel.setText(ReviewI18n.getString("ReviewIdNewReviewIdPage.label.reviewId"));
111        this.reviewIdText = new Text(parent, SWT.SINGLE | SWT.BORDER);
112        FormData reviewIdLabelData = new FormData();
113        reviewIdLabelData.top = new FormAttachment(reviewIdText, 0, SWT.CENTER);
114        reviewIdLabel.setLayoutData(reviewIdLabelData);
115        FormData reviewIdTextData = new FormData();
116        reviewIdTextData.top = new FormAttachment(infoLabel, 5);
117        reviewIdTextData.left = new FormAttachment(reviewIdLabel, 40);
118        reviewIdTextData.right = new FormAttachment(100, 0);
119        reviewIdText.setLayoutData(reviewIdTextData);
120      }
121      
122      /**
123       * Creates description content.
124       * @param parent the parent.
125       */
126      private void createDescriptionContent(Composite parent) {
127        Label descriptionLabel = new Label(parent, SWT.NONE);
128        descriptionLabel.setText(ReviewI18n.getString("ReviewIdNewReviewIdPage.label.description"));
129        this.descriptionText = new Text(parent, SWT.SINGLE | SWT.BORDER);
130        FormData descriptionLabelData = new FormData();
131        descriptionLabelData.top = new FormAttachment(descriptionText, 0, SWT.CENTER);
132        descriptionLabel.setLayoutData(descriptionLabelData);
133        FormData descriptionTextData = new FormData();
134        descriptionTextData.top = new FormAttachment(reviewIdText, 5);
135        descriptionTextData.left = new FormAttachment(reviewIdText, 0, SWT.LEFT);
136        descriptionTextData.right = new FormAttachment(100, 0);
137        descriptionText.setLayoutData(descriptionTextData);
138      }
139    
140      /**
141       * Handles project selection 
142       * @param e the event.
143       */
144      protected void handleProjectSelection(Event e) {
145        String projectName = ((Combo) e.widget.getData()).getText();
146        IProject project = FileResource.getProject(projectName);
147        PropertyResource reviewIdResource = PropertyResource.getInstance(this.project, false);
148        this.tempReviewIdList = Arrays.asList(reviewIdResource.getReviewIdNames());
149      }
150      
151      /**
152       * Adds listeners.
153       */
154      private void addListeners() {
155        this.reviewIdText.addListener(SWT.KeyUp, new ListenerAdapter());
156        this.descriptionText.addListener(SWT.KeyUp, new ListenerAdapter());
157      }
158      
159      /**
160       * Checks if the wizard can flip to the next page.
161       * @return <code>true</code> if it can flip to the next page.
162       */
163      public boolean canFlipToNextPage() {
164        return isPageComplete();
165      }
166    
167      /**
168       * Checks if the review id is valid (i.e. not duplicated).
169       * @param reviewId the review id.
170       * @return <code>true</code> if the review id is valid. <code>false</code> otherwise.
171       */
172      private boolean isValidReviewId(String reviewId) {
173        if (this.tempReviewIdList.contains(reviewId)) {
174          String message = ReviewI18n.getString("ReviewIdNewReviewIdPage.label.waring.idDuplicate");
175          applyToStatusLine(new Status(IStatus.WARNING, "not_used", 0, message , null));
176          return false;
177        }
178        else {
179          String message = ReviewI18n.getString("ReviewIdNewReviewIdPage.label.title.description");
180          applyToStatusLine(new Status(IStatus.OK, "not_used", 0, message , null));
181          return true;
182        }
183      }
184    
185      /**
186       * Checks if the text is non empty. Returns <code>true</code> if the text is not empty. Returns
187       * <code>false</code> otherwise.
188       * 
189       * @param text the text to be check.
190       * @return <code>true</code> if the text is not empty. Returns <code>false</code> otherwise.
191       */
192      private static boolean isTextNonEmpty(Text text) {
193        String string = text.getText();
194        if ((string != null) && (string.trim().length() > 0)) {
195          return true;
196        }
197        return false;
198      }
199    
200      /**
201       * Returns the next page. Saves the values from this page in the model associated with the wizard.
202       * Initializes the widgets on the next page.
203       * @return the next page.
204       */
205      public IWizardPage getNextPage() {
206        return ((ReviewIdNewWizard) getWizard()).getPage(ReviewIdNewWizard.PAGE_FILE);
207      }
208      
209      /**
210       * Gets the review id.
211       * @return the review id.
212       */
213      public String getReviewId() {
214        return this.reviewIdText.getText();
215      }
216      
217      /**
218       * Gets the description.
219       * @return the description.
220       */
221      public String getDescription() {
222        return this.descriptionText.getText();
223      }
224      
225      /**
226       * Applies the status to the status line of a dialog page.
227       * @param status the status.
228       */
229      private void applyToStatusLine(IStatus status) {
230        String message = status.getMessage();
231        if (message.length() == 0) {
232          message = null;
233        }
234        switch (status.getSeverity()) {
235          case IStatus.OK:
236            setErrorMessage(null);
237            setMessage(message);
238            break;
239          case IStatus.WARNING:
240            setErrorMessage(null);
241            setMessage(message, WizardPage.WARNING);
242            break;        
243          case IStatus.INFO:
244            setErrorMessage(null);
245            setMessage(message, WizardPage.INFORMATION);
246            break;      
247          default:
248            setErrorMessage(message);
249            setMessage(null);
250            break;    
251        }
252      }
253      
254      /**
255       * Provides a concrete listener adapter class.
256       * 
257       * @author Takuya Yamashita
258       * @version $Id: ReviewIdNewReviewIdPage.java,v 1.4 2005/03/25 18:32:06 takuyay Exp $
259       */
260      private class ListenerAdapter implements Listener {
261    
262        /**
263         * Handles event
264         * @see org.eclipse.swt.widgets.Listener#handleEvent(org.eclipse.swt.widgets.Event)
265         */
266        public void handleEvent(Event event) {
267          setPageComplete(isTextNonEmpty(reviewIdText) && isTextNonEmpty(descriptionText));
268          getWizard().getContainer().updateButtons();
269        }
270      }
271    }
