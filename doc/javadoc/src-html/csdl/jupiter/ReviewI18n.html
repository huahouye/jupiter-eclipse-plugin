001    package csdl.jupiter;
002    
003    import java.text.MessageFormat;
004    import java.util.Enumeration;
005    import java.util.HashMap;
006    import java.util.Locale;
007    import java.util.Map;
008    import java.util.MissingResourceException;
009    import java.util.ResourceBundle;
010    
011    /**
012     * Provides resources from resource bundle.
013     *
014     * @author Takuya Yamashita
015     * @version $Id: ReviewI18n.java,v 1.3 2005/03/25 18:32:09 takuyay Exp $
016     */
017    public class ReviewI18n {
018      /** the resource bundle file name */
019      private static final String RESOURCE_BUNDLE = "jupiter";
020      /** the resource bundle instance. */
021      private static ResourceBundle bundle = ResourceBundle.getBundle(RESOURCE_BUNDLE);
022      /** the value-key map.*/
023      private static Map valueKeyMap;
024      
025      static {
026        createValueKeyMap();
027      }
028    
029      /**
030       * Prevents clients from instantiating this class.
031       */
032      private ReviewI18n() {
033      }
034      
035      /**
036       * Creates value-key map. Used to get key from value.
037       */
038      private static void createValueKeyMap() {
039        valueKeyMap = new HashMap();
040        for (Enumeration keys = bundle.getKeys(); keys.hasMoreElements();) {
041          String key = (String) keys.nextElement();
042          String value = getString(key);
043          valueKeyMap.put(value, key);
044        }
045      }
046    
047      /**
048       * Returns the formatted message for the given key in the resource bundle.
049       *
050       * @param key the resource name
051       * @param args the message arguments
052       *
053       * @return the string
054       */
055      public static String format(String key, Object[] args) {
056        return MessageFormat.format(getString(key), args);
057      }
058    
059      /**
060       * Returns the string from the plugin's resource bundle,
061       * or 'key' if not found.
062       * @param key the key to search value in the resource bundle.
063       * @return the string from the plugin's resource bundle,
064       * or 'key' if not found.
065       */
066      public static String getString(String key) {
067        try {
068          return bundle.getString(key);
069        } 
070        catch (MissingResourceException e) {
071          return key;
072        }
073      }
074      
075      /**
076       * Gets the key from the value, or 'value' if not found.
077       * @param value the value to search value.
078       * @return the key from the value, or 'value' if not found.
079       */
080      public static String getKey(String value) {
081        String key = (String) valueKeyMap.get(value);
082        return (key != null) ? key : value;
083      }
084      
085      /**
086       * Sets locale for the resource bundle.
087       * @param locale the locale.
088       */
089      public void setLocale(Locale locale) {
090        bundle = ResourceBundle.getBundle(RESOURCE_BUNDLE, locale);
091        createValueKeyMap();
092      }
093    }
